	.TITLE	DBOXIO.TEXT; Cet extrait du logiciel DESSIN.CODE montre comment utiliser les; appels DBOX_PREND, DBOX_SAUVE, etc. pour les touches (F0), (F1),; (F2) et (F3).LBUFILE	=	50		; lg standard des noms de fichiers !BDBAUTO	=	15		; mode de sauvetage automatiqueBDBDEF	=	14		; sauve les dfinitionsBDBNMAJ	=	13		; touche (MET A JOUR) grise;-----------------------;;	touche (F0)	;;-----------------------;FINI:	LOAD.W	D3,(A6)+ONOZO	; D3 <-- zone actuelle	CALL	CLOZONE		; sauve la zone actuelle	LOAD.W	D4,D3		; D4 <-- zone actuelle	CLR.W	D3		; D3 <-- premire zone  sauver20$:	CALL	OPEZONE		; prend une zone	LOAD.W	(A6)+ONOZO,D3	CLR.W	D0		; D0 <-- avant de quitter	CALL	DBOXSAVE	; sauve si ncessaire avant de quitter	JUMP,EQ 50$		; ok => 50$	COMP.W	D7,#1		; (ANNULE) ?	JUMP,EQ 80$		; oui => 80$				; erreur =>	PUSH.W	D7	LOAD.W	D3,D4	CALL	OPEZONE		; reprend la zone initiale	LOAD.W	(A6)+ONOZO,D3	POP.W	D7	JUMP	ERROR50$:	INC.W	D3		; D3 <-- numro zone suivante	COMP.W	D3,#NBZONE	JUMP,LO 20$	CALL	DBOXQUITTE	; demande s'il faut quitter ...	COMP.W	D3,#F1	JUMP,EQ QUITTE80$:	LOAD.W	D3,D4	CALL	OPEZONE		; reprend la zone initiale	LOAD.W	(A6)+ONOZO,D3	JUMP	LOOPQUITTE:	...	LIB	?CLOLIB	FOS	?STOP;-----------------------;;	touche (F1)	;;-----------------------;PREND:	CLR.B	(A6)+OBUFINPRENDL:	LOAD.L	A3,#(A6)+OBUFIN	CALL	DBOXEDITI	JUMP,NE ERROR	COMP.W	D3,#F0	JUMP,EQ LOOP	LOAD.W	D0,#1		; D0 <-- avant de reprendre	CALL	DBOXSAVE	; sauve si ncessaire le dessin actuel	JUMP,EQ PRENDF		; ok => PRENDF	COMP.W	D7,#1		; (ANNULE) ?	JUMP,EQ LOOP		; oui => LOOP	JUMP	ERROR		; non => ERRORPRENDF:	CALL	WAITMOUSE	; met le sablier	LOAD.L	A4,#R16"TEXIM	; A4 <-- ^extension	CALL	ADDEXT		; A3 <-- ^nom du fichier .IMAGE	LOAD.W	D3,#2^BOPRD	LOAD.L	D0,#FOS?OPEN	CALL	ERCALL		; ouvre l'image	JUMP,EQ 20$	COMP.W	D7,#ERKSTO	JUMP,EQ LOOP	PUSH.W	D7	CLR.B	D3	CALL	SELMO		; outil = flche	CALL	SCMO		; remet la flche	CALL	NORMMOUSE	; remet la flche	LOAD.W	D3,#BIPERR	LIB	?BEEP		; fait un bruit	POP.W	D7	LOAD.W	D2,#2^BDBSAV	; D2 <-- mode	DBOX_	MERROR		; affiche l'erreur	JUMP	PRENDL20$:	...	FOS	?RDBYTE		; lit l'en-tte	...	FOS	?CLOSE		; ferme l'image	...	LOAD.L	A3,#(A6)+OBUFIN	CALL	SAVFILE		; OBUNAME <-- nom	CALL	DESINFFILE	; affiche le nom dans les infos	CLR.B	(A6)+OMAJ	; le dessin est  jour	JUMP	LOOP;-----------------------;;	touche (F2)	;;-----------------------;AJOUTE:	LOAD.L	A3,#(A6)+OBUFAJ	CALL	DBOXEDITA	JUMP,NE ERROR	COMP.W	D3,#F0	JUMP,EQ LOOP	CALL	WAITMOUSE	; met le sablier	LOAD.L	A4,#R16"TEXIM	; A4 <-- ^extension	CALL	ADDEXT		; A3 <-- ^nom du fichier .IMAGE	LOAD.W	D3,#2^BOPRD	LOAD.L	D0,#FOS?OPEN	CALL	ERCALL		; ouvre l'image	JUMP,EQ 20$	COMP.W	D7,#ERKSTO	JUMP,EQ LOOP	PUSH.W	D7	CLR.B	D3	CALL	SELMO		; outil = flche	CALL	SCMO		; remet la flche	CALL	NORMMOUSE	; remet la flche	LOAD.W	D3,#BIPERR	LIB	?BEEP		; fait un bruit	POP.W	D7	LOAD.W	D2,#2^BDBSAV	; D2 <-- mode	DBOX_	MERROR		; affiche l'erreur	JUMP	AJOUTE20$:	...	FOS	?RDBYTE		; lit l'en-tte	...	FOS	?CLOSE		; ferme l'image	...	JUMP	LOOP;-----------------------;;	touche (F3)	;;-----------------------;SAUVE:	CLR.W	D2		; D2 <-- mode manuel	CALL	SAUVETOUT	; sauve toute l'image	JUMP,NE ERROR	JUMP	LOOP;----------\\; SAUVETOUT >;==========/; Sauve toute l'image.; in	D2.W	mode; out	D7.W	erreur; mod	D7.WSAUVETOUT:	PUSHM.L	D0..D5,A3,A4	LOAD.L	A3,#(A6)+OBUFOUT	CALL	GETFILE		; OBUFOUT <-- OBUNAME	CLR.L	D5	TEST.B	(A6)+OMAJ	; mise  jour ncessaire ?	JUMP,F	10$		; non => 10$				; oui =>	CALL	IFVIDE		; dessin vide ?	JUMP,NE 11$		; non => 11$10$:	TSET.L	D2:#BDBNMAJ	; D2 <-- touche (MET A JOUR) grise11$:	LOAD.L	A3,#(A6)+OBUFOUT	LOAD.L	D4,D5		; D4 <-- fanions	CALL	DBOXEDITO	JUMP,NE 95$	COMP.W	D3,#F0	JUMP,EQ 95$	LOAD.W	D1,D3		; D1 <-- F1/F4	LOAD.L	D5,D4		; D5 <-- fanions	CALL	WAITMOUSE	; met le sablier	LOAD.L	A4,#R16"TEXIM	; A4 <-- ^extension	CALL	ADDEXT		; A3 <-- ^nom du fichier avec extension20$:	LOAD.L	D3,#DEFATT	; D3 <-- attributs par dfaut	LOAD.W	D4,#TYPRAND	; D4 <-- type "random"	LOAD.L	D0,#FOS?CREATE	CALL	ERCALL		; cre le fichier	JUMP,EQ 50$		; ok => 50$	COMP.W	D7,#ERFAEX	; fichier existe dj ?	JUMP,NE	90$		; non => 90$				; oui =>	TEST.L	D2:#BDBAUTO	; sauvetage automatique ?	JUMP,BS 40$		; oui => 40$	COMP.W	D1,#F4		; touche (MET A JOUR) ?	JUMP,EQ 40$		; oui => 40$	CALL	NORMMOUSE	; remet la flche	PUSH.L	D2	LOAD.W	D2,#2^BDBSAV	; D2 <-- mode	LIST_	DELETE		; dtruit le fichier ?	POP.L	D2	PUSH.W	D7	CALL	AFMENU		; refait les soft-keys	POP.W	D7	JUMP,NE 90$	COMP.W	D3,#'O	JUMP,NE 10$40$:	LOAD.L	D0,#FOS?DELETE	CALL	ERCALL		; dtruit le fichier	JUMP,NE 90$	JUMP	20$50$:	CALL	WAITMOUSE	; met le sablier	...	FOS	?WRBYTE	...	FOS	?CLOSE	...90$:	TEST.W	D7		; ok =	JUMP,NE 95$		; non => 90$	LOAD.L	A3,#(A6)+OBUFOUT	CALL	SAVFILE		; OBUNAME <-- nom	CALL	DESINFFILE	; affiche le nom dans les infos	CLR.B	(A6)+OMAJ	; le dessin est  jour	CLR.W	D7		; D7 <-- ok95$:	POPM.L	D0..D5,A3,A4	TEST.W	D7		; retour EQ/NE	RET;----------\\; DBOXEDITO >;==========/; Edite un nom de fichier en sortie.; in	A3.L	^nom du fichier;	D4.L	fanions par dfaut;	D2.W	mode; out	D2.W	mode;	D3.W	dernire touche (F1/F4/F0);	D7.W	erreur; mod	D2.L, D3.L, D7.WDBOXEDITO:	PUSHM.L	D1,D4,A2..A5	LOAD.W	D3,#F1		; D3 <-- comme si (SAUVE)	TEST.L	D2:#BDBAUTO	; mode automatique ?	JUMP,BS	90$		; oui => 90$	CALL	DBOXOPEN1	LOAD.W	D4,D2		; D4 <-- fanions	LOAD.L	A5,A3		; A5 <-- ^nom du fichier	LOAD.L	A2,#R16"TDBOX_SAUVE	LOAD.L	A3,#R16"TDBOX_IO	LOAD.L	A4,#R16"TEXIM	; A4 <-- ^critre de recherche (extension)	LOAD.W	D1,#DBOX_SAUVE	; D1 <-- code du sous-appel	TSET.L	D2:#BDBEXTEND	; D2 <-- sous-appel d'extension	TSET.L	D2:#BDBSAV	; D2 <-- sauve le fond	LOAD.L	D3,#LBUFILE!2^BDBIOOBJ!2^BDBIOEXT1	TEST.L	D2:#BDBNMAJ	JUMP,BC 30$	TSET.L	D3:#BDBIONMAJ	; D3 <-- touche (MET A JOUR) grise30$:	DBOX_	EXTEND		; dbox: attend la rponse ...	LOAD.W	D2,D4		; D2 <-- nouveau mode (fanions)	PUSH.W	D7	CALL	DBOXCLOSE1	POP.W	D790$:	POPM.L	D1,D4,A2..A5	TEST.W	D7		; retour EQ/NE	RET;----------\\; DBOXEDITE >;==========/; Edite un nom de fichier en sortie (extraction).; in	A3.L	^nom du fichier;	D4.L	fanions par dfaut; out	D3.W	dernire touche (F1/F0);	D7.W	erreur; mod	D3.L, D7.WDBOXEDITE:	PUSHM.L	D1,D2,D4,A2..A5	CALL	DBOXOPEN1	CLR.L	D4		; D4 <-- fanions	LOAD.L	A5,A3		; A5 <-- ^nom du fichier	LOAD.L	A2,#R16"TDBOX_SAUVE	LOAD.L	A3,#R16"TDBOX_IO	LOAD.L	A4,#R16"TEXIM	; A4 <-- ^critre de recherche (extension)	LOAD.W	D1,#DBOX_EXTRAIT; D1 <-- code du sous-appel	LOAD.W	D2,#2^BDBSAV!2^BDBEXTEND	LOAD.L	D3,#LBUFILE!2^BDBIOOBJ!2^BDBIOEXT1	DBOX_	EXTEND		; dbox: attend la rponse ...	LOAD.W	D2,D4		; D2 <-- nouveau mode (fanions)	PUSH.W	D7	CALL	DBOXCLOSE1	POP.W	D790$:	POPM.L	D1,D2,D4,A2..A5	TEST.W	D7		; retour EQ/NE	RET;----------\\; DBOXEDITI >;==========/; Edite un nom de fichier en entre (prend).; in	A3.L	^nom du fichier; out	D3.W	dernire touche (F1/F0);	D4.L	tats des fanions;	D7.W	erreur; mod	D3.L, D4.L, D7.WDBOXEDITI:	PUSHM.L	D1,D2,A2..A5	CALL	DBOXOPEN1	CLR.L	D4		; D4 <-- fanions	LOAD.L	A5,A3		; A5 <-- ^nom du fichier	LOAD.L	A3,#R16"TDBOX_IO	LOAD.L	A4,#R16"TEXIM	; A4 <-- ^critre de recherche (extension)	LOAD.W	D1,#DBOX_PREND	; D1 <-- code du sous-appel	LOAD.W	D2,#2^BDBSAV!2^BDBEXTEND	LOAD.L	D3,#LBUFILE!2^BDBIOEXT1	DBOX_	EXTEND		; dbox: attend la rponse ...	PUSH.W	D7	CALL	DBOXCLOSE1	POP.W	D790$:	POPM.L	D1,D2,A2..A5	TEST.W	D7		; retour EQ/NE	RET;----------\\; DBOXEDITA >;==========/; Edite un nom de fichier en entre (ajoute).; in	A3.L	^nom du fichier; out	D3.W	dernire touche (F1/F0);	D4.L	tats des fanions;	D7.W	erreur; mod	D3.L, D4.L, D7.WDBOXEDITA:	PUSHM.L	D1,D2,A2..A5	CALL	DBOXOPEN1	CLR.L	D4		; D4 <-- fanions	LOAD.L	A5,A3		; A5 <-- ^nom du fichier	LOAD.L	A3,#R16"TDBOX_IO	LOAD.L	A4,#R16"TEXIM	; A4 <-- ^critre de recherche (extension)	LOAD.W	D1,#DBOX_AJOUTE	; D1 <-- code du sous-appel	LOAD.W	D2,#2^BDBSAV!2^BDBEXTEND	LOAD.L	D3,#LBUFILE!2^BDBIOEXT1	DBOX_	EXTEND		; dbox: attend la rponse ...	PUSH.W	D7	CALL	DBOXCLOSE1	POP.W	D790$:	POPM.L	D1,D2,A2..A5	TEST.W	D7		; retour EQ/NE	RET;---------\\; DBOXSAVE >;=========/; Sauve le dessin selon le choix de l'utilisateur.; in	D0.W	mode (0=quitte, 1=nouveau texte); out	D7.W	0	=> ok, on peut quitter;		1	=> (ANNULE), donc on reste;		autre	=> erreur grve; mod	D7.WDBOXSAVE:	PUSHM.L	D1..D4,A2..A5	TEST.B	(A6)+OMAJ	; mise  jour ncessaire ?	JUMP,F	80$		; non => 80$ (quitte, snif)				; oui =>	CALL	IFVIDE		; dessin vide ?	JUMP,NE 10$		; non => 10$				; oui =>	CLR.B	(A6)+OMAJ	; fait comme si dessin  jour	JUMP	80$10$:	CALL	DBOXOPEN1	CLR.L	D4		; D4 <-- fanions	CLR.L	D3	LOAD.W	D3,(A6)+ONOZO	INC.B	D3		; D3 <-- numro de la page  afficher	COMP.B	D3,#NBZONE	JUMP,NE 20$	CLR.B	D320$:	TEST.B	D0	JUMP,NE 25$	TSET.L	D3:#BDBIOQUIT	; D3 <-- texte "avant de quitter"25$:	LOAD.L	A5,#(A6)+OBUNAME; A5 <-- ^nom du fichier	LOAD.L	A2,#R16"TDBOX_SAUVE	LOAD.L	A3,#R16"TDBOX_IO	LOAD.W	D1,#DBOX_MAJ	; D1 <-- code du sous-appel	LOAD.W	D2,#2^BDBSAV!2^BDBEXTEND	TSET.L	D3:#BDBIOOBJ	DBOX_	EXTEND		; dbox: attend la rponse ...	PUSH.W	D7	CALL	DBOXCLOSE1	POP.W	D7	COMP.W	D3,#F0		; ANNULE ?	JUMP,EQ 85$	COMP.W	D3,#F4		; NON ?	JUMP,EQ 80$	OR.W	D2,D4		; D2 <-- met aussi les fanions (pour BDBDEF)	CALL	SAUVETOUT	; sauve tout sur disque (in D2.W !)	JUMP	90$80$:	CLR.W	D7		; D7 <-- quitte	JUMP	90$85$:	LOAD.W	D7,#1		; D7 <-- reste90$:	POPM.L	D1..D4,A2..A5	TEST.W	D7		; retour EQ/NE	RET;-----------\\; DBOXQUITTE >;===========/; Demande si l'on veut quitter DESSIN.; in	-; out	D3.W	dernire touche (F1/F4);	D7.W	erreur; mod	D3.L, D7.WDBOXQUITTE:	PUSHM.L	D1,D2,D4,A2..A5	CALL	DBOXOPEN1	CLR.L	D4		; D4 <-- fanions	LOAD.L	A3,#R16"TDBOX_IO	LOAD.W	D1,#DBOX_QUITTE	; D1 <-- code du sous-appel	LOAD.W	D2,#2^BDBSAV!2^BDBEXTEND	CLR.L	D3	DBOX_	EXTEND		; dbox: attend la rponse ...	PUSH.W	D7	CALL	DBOXCLOSE1	POP.W	D790$:	POPM.L	D1,D2,D4,A2..A5	TEST.W	D7		; retour EQ/NE	RET;-----------\\; DBOXEFFACE >;===========/; Demande s'il faut effacer le dessin.; in	-; out	D3.W	dernire touche (F1/F4);	D7.W	erreur; mod	D3.L, D7.WDBOXEFFACE:	PUSHM.L	D1,D2,D4,A2..A5	CALL	DBOXOPEN1	CLR.L	D4		; D4 <-- fanions	CLR.L	D3	LOAD.W	D3,(A6)+ONOZO	INC.B	D3		; D3 <-- numro de la page  afficher	COMP.B	D3,#NBZONE	JUMP,NE 20$	CLR.B	D320$:	LOAD.L	A5,#(A6)+OBUNAME; A5 <-- ^nom du fichier	LOAD.L	A3,#R16"TDBOX_IO	LOAD.W	D1,#DBOX_EFFACE	; D1 <-- code du sous-appel	LOAD.W	D2,#2^BDBSAV!2^BDBEXTEND	DBOX_	EXTEND		; dbox: attend la rponse ...	PUSH.W	D7	CALL	DBOXCLOSE1	POP.W	D7	POPM.L	D1,D2,D4,A2..A5	TEST.W	D7		; retour EQ/NE	RET;---------\\; DBOXOPEN >;---------/; Ouvre une dbox.; in	A2.L	table;	D2.W	mode; out	D5.L	canal;	D7.W	erreur; mod	D5.L, D7.WDBOXOPEN:	CALL	DBOXOPEN1	DBOX_	OPEN		; ouvre la dbox	RETDBOXOPEN1:	PUSH.W	D3	CALL	ERGMOUSE	; efface les flches dans les rgles	CLR.B	D3	CALL	SELMO		; outil = flche	CALL	SCMO		; remet la flche	CALL	NORMMOUSE	; remet la flche	POP.W	D3	RET;----------\\; DBOXCLOSE >;----------/; Ferme une dbox.; in	D5.L	canal; out	D7.W	erreur; mod	D7.WDBOXCLOSE:	DBOX_	CLOSE		; ferme la dboxDBOXCLOSE1:	CALL	AFMENU		; refait les soft-keys	RET; Tables; ------TDBOX_SAUVE:	.B	LDBOX_SAUVE	.B	ODBSQUARE,BDBDEF	.ASCIZ	" sauve les pinceaux et les trames"	.B	ODBCRLDBOX_SAUVE	=	.-TDBOX_SAUVE-1TDBOX_IO:	.ASCIZ	"DESSIN"	; nom du logiciel	.ASCIZ	"Fimage"	; [M,F] nom des lments manipuls	.ASCIZ	"Fpage"		; [M,F] nom du contenant	.B	0	.EVEN	.END
