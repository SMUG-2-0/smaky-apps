	.TITLE	ERCALL.TEXTLes instructions :	LOAD.W	D3,#2^BOPRD	FOS	?OPEN		; ouvre le fichier	JUMP,NE ERRORdoivent tre remplaces par :	LOAD.W	D3,#2^BOPRD	PUSH.L	D0	LOAD.L	D0,#FOS?OPEN	CALL	ERCALL		; ouvre le fichier	POPM.L	D0	JUMP,NE ERRORet ceci pour les appels FOS suivants :	FOS	?CREATE	FOS	?OPEN	FOS	?DELETE	FOS	?RENAME	FOS	?CHATR	FOS	?ARGS;--------\\; ERCALL  >;========/; Excute un appel systme en essayant de rsoudre qq problmes.; in	D0.L	code de l'appel (par exemple #FOS?OPEN);	+ selon appel; out	+ selon appel;	D7.W	erreur (ERKSTO si annule); mod	D7.WERCALL:	PUSHM.L	D1,D2	LOAD.W	D1,#DBOX_ERCALL	; D1 <-- numro de l'appel	LOAD.W	D2,#2^BDBEXTEND	; D2 <-- mode "tendu"	DBOX_	EXTEND		; appel DBOX_ERCALL ...	POPM.L	D1,D2	TEST.W	D7		; retour EQ/NE	RET----------->	NE PAS OUBLIER :ERROR:	COMP.W	D7,#ERKSTO	; touche (ANNULE) utilise ?	JUMP,EQ LOOP		; oui => LOOP	CALL	AFERROR		; non => affiche l'erreur	JUMP	LOOP	.END
